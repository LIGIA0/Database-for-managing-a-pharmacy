create table CLIENTI_PROIECT
(
  ID_CLIENT           NUMBER(6) not null,
  prenume_client      VARCHAR2(20),
  nume_client         VARCHAR2(20),
  telefon_client      VARCHAR2(10) CONSTRAINT CLIENTI_TELEFON_UQ UNIQUE,
  email_client        VARCHAR2(30),
  data_nastere        DATE,
  sex                 VARCHAR2(1)
);
alter table CLIENTI_PROIECT
  add constraint CLIENTI_PROIECT_ID_CLIENT_PK primary key (ID_CLIENT);
alter table CLIENTI_PROIECT
  add constraint CLIENTI_PROIECT_NUME_NN
  check ("NUME_CLIENT" IS NOT NULL);
alter table CLIENTI_PROIECT
  add constraint CLIENTI_PROIECT_PRENUME_NN
  check ("PRENUME_CLIENT" IS NOT NULL);
alter table CLIENTI_PROIECT
  add constraint CLIENTI_PROIECT__EMAIL_CK CHECK (email_client LIKE '%@%.%');

create table RETETE_PROIECT
(
  COD_RETETA        NUMBER(12) not null,
  data_eliberarii   DATE,
  tip_reteta        VARCHAR2(20),
  diagnostic        VARCHAR2(20),
  emitent           VARCHAR2(30),
  nume_emitent      VARCHAR2(20),
  ID_CLIENT         NUMBER(6) not null,
constraint RETETE_DIAGNOSTIC_NN check (diagnostic IS NOT NULL),
constraint RETETE_TIP_RETETA_CK check (tip_reteta in ('compensata','necompensata','gratuita'))
);
alter table RETETE_PROIECT
  add constraint RETETE_COD_RETETA_PK primary key (COD_RETETA);
alter table RETETE_PROIECT
  add constraint RETETE_ID_CLIENT_FK foreign key (ID_CLIENTI)
  references CLIENTI_PROIECT (ID_CLIENT) on delete set null;
alter table RETETE_PROIECT
  add constraint RETETE_DATA_ELIBERARII_NN
  check ("DATA_ELIBERARII" IS NOT NULL);
alter table RETETE_PROIECT
   add constraint RETETE_NUME_EMITENT_NN
   check ("NUME_EMITENT" IS NOT NULL);

create table MEDICAMENTE_PROIECT
(
 COD_MEDICAMENT           NUMBER(8) not null CONSTRAINT medicamente_cod_medicament_pk PRIMARY KEY,
 pret                     NUMBER(8,2),
 stoc                     NUMBER(5),
 natura_excipientului     VARCHAR(30) NOT NULL,
 producator               VARCHAR(40) NOT NULL,
 denumire_medicament      VARCHAR(40) NOT NULL
);

create table DETALII_RETETE_PROIECT
(
 COD_MEDICAMENT NUMBER(8) constraint DETALII_RETETE_FK references MEDICAMENTE_PROIECT(COD_MEDICAMENT),
 dozaj          NUMBER,
 ID_RETETA      NUMBER(12),
constraint RETETA_PROIECT_FK foreign key (ID_RETETA) references RETETE_PROIECT(COD_RETETA),
constraint DETALII_RETETA_PK primary key (COD_MEDICAMENT, ID_RETETA)
);

create table VANZARI_PROIECT
(
 COD_VANZARE       NUMBER(8) constraint VANZARI_PROIECT_PK primary key,
 cantitate         NUMBER(3) not null,
 ID_CLIENT         NUMBER(6),
 COD_RETETA        NUMBER(12),
 COD_MEDICAMENT    NUMBER(8),
constraint FK_COD_RETETA foreign key (COD_RETETA) references RETETE_PROIECT(COD_RETETA),
constraint FK_ID_CLIENT foreign key (ID_CLIENT) references CLIENTI_PROIECT(ID_CLIENT),
constraint FK_COD_MEDICAMENT foreign key(COD_MEDICAMENT) references MEDICAMENTE_PROIECT(COD_MEDICAMENT)
);


create table ANGAJATI_PROIECT
(
  ID_ANGAJATI           NUMBER(6) not null,
  prenume_angajat       VARCHAR2(20) not null,
  nume_angajat          VARCHAR2(25) not null,
  email_angajat         VARCHAR2(25),
  telefon_angajat       VARCHAR2(20),
  data_angajare         DATE,
  salariul_angajat      NUMBER(8,2),
  comision              NUMBER(2,2)
);
alter table ANGAJATI_PROIECT
  add constraint ANG_ID_ANGAJATI_PK primary key (ID_ANGAJATI);
alter table ANGAJATI_PROIECT
  add constraint ANG_EMAIL_UKKK unique (EMAIL_ANGAJAT);

create table BONURI_PROIECT
(
 NR_BONURI          NUMBER(8) constraint BONURI_PROIECT_PK primary key,
 data_vanzare    DATE,    
 COD_VANZARE     NUMBER(8) constraint FK_COD_VANZARE references VANZARI_PROIECT(COD_VANZARE),
 ID_ANGAJATI     NUMBER(6) constraint ID_ANGAJATI references ANGAJATI_PROIECT(ID_ANGAJATI)
);
